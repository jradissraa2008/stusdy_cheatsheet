SQL Cheatsheet

LDD · LMD · Contraintes · Requêtes — tout ce qu'il faut pour le bac 🌸

CREATE — Créer
Base de données
CREATE DATABASE NomBD ;
Table simple
CREATE TABLE NomTable ( col1 INT PRIMARY KEY, col2 VARCHAR(30) NOT NULL, col3 DECIMAL(8,3) ) ;
Avec clé étrangère
col INT FOREIGN KEY REFERENCES AutreTable(col) ON UPDATE CASCADE ON DELETE CASCADE
Clé primaire composée
PRIMARY KEY(col1, col2)
Contraintes CHECK
Email contient @
CHECK(email LIKE '%@%')
Valeur entre deux bornes
CHECK(age BETWEEN 1 AND 80) -- ou CHECK(age >= 1 AND age <= 80)
Valeur dans une liste
CHECK(Genre IN ('H', 'F'))
Date DLC (année+4)
CHECK(YEAR(dlc) = YEAR(NOW()) + 4 AND MONTH(dlc) = MONTH(NOW()))
Default + Check combiné
QteStock INT DEFAULT 0 CHECK(QteStock > 0)
ALTER TABLE — Modifier la structure
Ajouter une colonne
ALTER TABLE T ADD col INT ;
Modifier type/taille
ALTER TABLE T MODIFY col VARCHAR(30) ;
Supprimer une colonne
ALTER TABLE T DROP col ;
Renommer une colonne
ALTER TABLE T RENAME ancienNom TO nouveauNom ;
Valeur par défaut
ALTER TABLE T MODIFY col DATE DEFAULT NOW() ;
Ajouter clé primaire
ALTER TABLE T ADD col CHAR(5) PRIMARY KEY ;
Ajouter clé étrangère
ALTER TABLE T ADD FOREIGN KEY (col) REFERENCES Autre(col) ;
Ajouter une contrainte
ALTER TABLE T ADD CONSTRAINT nomCk CHECK(condition) ;
Désactiver / Activer
ALTER TABLE T DISABLE CONSTRAINT nomCk ; ALTER TABLE T ENABLE CONSTRAINT nomCk ;
Supprimer une contrainte
ALTER TABLE T DROP CONSTRAINT nomCk ;
INSERT / UPDATE / DELETE
Insérer
INSERT INTO Table VALUES('val1', 'val2', ...) ;
Modifier
UPDATE Table SET col = 'nouvelle valeur' WHERE condition ;
Supprimer
DELETE FROM Table WHERE condition ;
SELECT — Requêtes
Jointure entre tables
SELECT col1, col2 FROM T1, T2 WHERE T1.id = T2.id AND condition ;
Trier
ORDER BY col DESC -- ou ASC
Fonctions d'agrégat
COUNT(col) -- nombre SUM(col) -- somme AVG(col) -- moyenne MAX(col) -- maximum MIN(col) -- minimum
Alias
SUM(PrixLog) AS CoutTotal

Supprimer table
DROP TABLE NomTable ;

col INT PRIMARY KEY AUTO_INCREMENT
Rappels clés 🎀
PRIMARY KEY = unique + NOT NULL FOREIGN KEY → intégrité référentielle NOT NULL → valeur obligatoire UNIQUE → pas de doublons DEFAULT → valeur par défaut CHECK → condition sur valeur CASCADE → propagation auto LIKE '%@%' → contient @ NOW() → date du jour BETWEEN x AND y → intervalle IN ('a','b') → liste de valeurs ORDER BY DESC → ordre décroissant
